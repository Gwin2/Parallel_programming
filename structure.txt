Структура проекта

├── docs/                  # Документация
│   ├── algorithm.md       # Описание алгоритма
│   └── results.md         # Результаты тестов
├── scripts/               # Вспомогательные скрипты
│   ├── build.sh           # Скрипт сборки
│   └── run_benchmarks.sh  # Скрипт бенчмарков
├── .github/               # CI/CD конфигурация

│       └── ci-cd.yml      # GitHub Actions
